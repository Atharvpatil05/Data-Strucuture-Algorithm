public class fibonaci_recursion {
    public static int fibo(int n) {
        if (n==0 || n==1){
            return n;
        }
        int fibosum=fibo(n-1)+fibo(n-2);
        return fibosum;
    }
    public static void main(String[] args) {
        int n=4;
        System.out.println(fibo(n));
    }
}
